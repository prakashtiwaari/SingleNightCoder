Create a child process so that parent writes data to pipe and child process read from that pipe

#include<stdio.h>
#include<pthread.h>
#include<unistd.h>
#include<sys/stat.h>
#include<stdlib.h>
#include<string.h>
#include<sys/wait.h>
#define max 100
int main()
{
int fd[2];
char buffer[100];
pid_t pid;
pipe(fd);
pid=fork();
if(pid<0)
{
printf(" fork Error\n");
return -1;
}
if(pid==0)
{
close(fd[1]);
read(fd[0],buffer,100);
close(fd[0]);

printf(" I am the child and i will be receiving a message from parent(%d)\n",getppid());
printf("The message is:- ");
puts(buffer);
}
if(pid>0)
{
printf("I am the parent and i will be writing to the child(%d)\n",pid);
printf("Enter the message\n");
char msg[max];
gets(msg);
close(fd[0]);
write(fd[1],msg,100);
close(fd[1]);
wait(NULL);
}
return 0;
}
}
